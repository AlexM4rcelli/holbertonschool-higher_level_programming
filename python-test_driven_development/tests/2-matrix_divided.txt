matrix_divided = __import__('2-matrix_divided').matrix_divided

# Test 1: Divide all elements of a 2x2 matrix by 2
assert matrix_divided([[4, 6], [8, 10]], 2) == [[2.0, 3.0], [4.0, 5.0]]

# Test 2: Divide all elements of a 3x3 matrix by 3.5
assert matrix_divided([[10, 15, 20], [25, 30, 35], [40, 45, 50]], 3.5) == [[2.86, 4.29, 5.71], [7.14, 8.57, 10.0], [11.43, 12.86, 14.29]]

# Test 3: Handling of non-integer/float input in the matrix
try:
    matrix_divided([[4, 6], [8, '10']], 2)
except TypeError as e:
    assert str(e) == "matrix must be a matrix (list of lists) of integers/floats"

# Test 4: Handling of rows with different sizes in the matrix
try:
    matrix_divided([[4, 6], [8, 10, 12]], 2)
except TypeError as e:
    assert str(e) == "Each row of the matrix must have the same size"

# Test 5: Handling of div equal to zero
try:
    matrix_divided([[4, 6], [8, 10]], 0)
except ZeroDivisionError as e:
    assert str(e) == "division by zero"
