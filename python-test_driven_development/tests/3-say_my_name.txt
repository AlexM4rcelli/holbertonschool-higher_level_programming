say_my_name = __import__('3-say_my_name').say_my_name

# Test 1: Print name with first name only
assert say_my_name("John") == "My name is John "

# Test 2: Print name with first name and last name
assert say_my_name("Jane", "Doe") == "My name is Jane Doe"

# Test 3: Print name with last name as an empty string
assert say_my_name("Alice") == "My name is Alice "

# Test 4: Handling of non-string input for first name
try:
    say_my_name(42, "Smith")
except TypeError as e:
    assert str(e) == "first_name must be a string"

# Test 5: Handling of non-string input for last name
try:
    say_my_name("Bob", 123)
except TypeError as e:
    assert str(e) == "last_name must be a string"
